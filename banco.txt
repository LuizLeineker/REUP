CREATE DATABASE reup_db;
USE reup_db;


CREATE TABLE produtos (
    id INT PRIMARY KEY,
    modelo VARCHAR(50) NOT NULL,
    tamanho VARCHAR(5) NOT NULL,
    marca VARCHAR(50) NOT NULL
);


CREATE USER 'usuario_reup'@'localhost' IDENTIFIED BY 'sua_senha';

GRANT ALL PRIVILEGES ON reup_db.* TO 'usuario_reup'@'localhost';

FLUSH PRIVILEGES;


SHOW TABLES;

SELECT * FROM produtos;
